✅ 1. ORGANIZATIONS
CREATE TABLE organizations (
    id VARCHAR(100) PRIMARY KEY,
    org_name VARCHAR(150) NOT NULL,
    plan_type VARCHAR(50),
    budget_limit DECIMAL(14,6),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
✅ 2. PROJECTS
CREATE TABLE projects (
    id VARCHAR(100) PRIMARY KEY,
    org_id VARCHAR(100) NOT NULL,
    project_name VARCHAR(150),
    environment VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (org_id) REFERENCES organizations(id)
);
✅ 3. USERS
CREATE TABLE users (
    id VARCHAR(100) PRIMARY KEY,
    org_id VARCHAR(100),
    email VARCHAR(150),
    role VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (org_id) REFERENCES organizations(id)
);
✅ 4. API_KEYS
CREATE TABLE api_keys (
    id VARCHAR(120) PRIMARY KEY,
    org_id VARCHAR(100),
    project_id VARCHAR(100),
    key_name VARCHAR(100),
    provider VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (org_id) REFERENCES organizations(id),
    FOREIGN KEY (project_id) REFERENCES projects(id)
);
✅ 5. USER_PROJECTS
CREATE TABLE user_projects (
    user_id VARCHAR(100),
    project_id VARCHAR(100),
    role VARCHAR(50),
    PRIMARY KEY (user_id, project_id),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (project_id) REFERENCES projects(id)
);
✅ 6. UPLOAD_DATA

    FOREIGN KEY (org_id) REFERENCES organizations(id),
    FOREIGN KEY (project_id) REFERENCES projects(id),

✅ 12. TELEMETRY_EVENTS (CORE TABLE)
CREATE TABLE telemetry_events (
    id BIGSERIAL PRIMARY KEY,
    event_id VARCHAR(120) UNIQUE NOT NULL,
    request_id VARCHAR(120),
    trace_id VARCHAR(120),
    org_id VARCHAR(100),
    project_id VARCHAR(100),
    user_id VARCHAR(100),
    api_key_id VARCHAR(120),
    provider VARCHAR(100),
    model_name VARCHAR(100),
    service_type VARCHAR(100),
    endpoint_name VARCHAR(150),
    input_tokens INT,
    output_tokens INT,
    total_tokens INT,
    latency_ms DECIMAL(10,2),
    cost DECIMAL(14,6),
    status VARCHAR(50),
    error_message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (org_id) REFERENCES organizations(id),
    FOREIGN KEY (project_id) REFERENCES projects(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
);
✅ 13. DATA_SECURITY_LOGS
CREATE TABLE data_security_logs (
    id BIGSERIAL PRIMARY KEY,
    pii_detected BOOLEAN,

    misuse_pattern_detected BOOLEAN,

    masking_applied BOOLEAN,
    risk_score DECIMAL(5,2),
    data_in_mb DECIMAL(10,2),
    data_out_mb DECIMAL(10,2),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
✅ 14. EXECUTION_PIPELINE
CREATE TABLE execution_pipeline (
    id BIGSERIAL PRIMARY KEY,

    stage_latency_ms DECIMAL(10,2),

    telemetry_id BIGINT,
    FOREIGN KEY (telemetry_id) REFERENCES telemetry_events(id)
);

✅ 16. USAGE_ANOMALIES
CREATE TABLE usage_anomalies (
    id BIGSERIAL PRIMARY KEY,
    org_id VARCHAR(100),
    tool_name VARCHAR(150) NOT NULL,
    event_id VARCHAR(120),
    anomaly_type VARCHAR(60) NOT NULL,
    severity VARCHAR(20) NOT NULL DEFAULT 'medium',
    anomaly_score DECIMAL(8,2) DEFAULT 0,
    baseline_value DECIMAL(14,6) DEFAULT 0,
    observed_value DECIMAL(14,6) DEFAULT 0,
    message TEXT,
    status VARCHAR(20) NOT NULL DEFAULT 'open',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

    misuse_count INT DEFAULT 0,
    total_input_mb DECIMAL(12,4) DEFAULT 0,
